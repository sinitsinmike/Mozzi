67 /** @ingroup midi
68 Converts midi note number to frequency. Caution: this can take up to 400
69 microseconds to run. It can seriously mess up the audio output if you use it in
70 updateControl() or updateAudio(). This is a good choice in setup(), or where you
71 need precise midi-pitch conversion and aren't doing much other audio
72 calculation.
73 @note Beware this returns an invalid result for midi note 0.
74 @note Timing: ~350 us
75 @param midival a midi note number, 1.0 or greater. Like the mtof object in Pd, midi values can have fractions.
76 @return the frequency represented by the input midi note number..
77  */
78 inline float mtof(float midival)
79 {
80  // http://en.wikipedia.org/wiki/Note
81  // f = pow(2,(p-69/12) * 440Hz
82  // return pow(2.0,(midival-69.0/12.0) * 440.0;
83 
84  // code from AF_precision_synthesis sketch, copyright 2009, Adrian Freed.
85  float f = 0.0;
86  if(midival) f = 8.1757989156 * pow(2.0, midival/12.0);
87  return f;
88 };

110 /** @ingroup midi
111 Converts midi note number to frequency with speed and accuracy. Q16n16_mtofLookup() is a fast
112 alternative to (float) mtof(), and more accurate than (uint8_t) mtof(),
113 using Q16n16 fixed-point format instead of floats or uint8_t values. Q16n16_mtof()
114 uses cheap linear interpolation between whole midi-note frequency equivalents
115 stored in a lookup table, so is less accurate than the float version, mtof(),
116 for non-whole midi values.
117 @note Timing: ~8 us.
118 @param midival_fractional a midi note number in Q16n16 format, for fractional values.
119 @return the frequency represented by the input midi note number, in Q16n16
120 fixed point fractional integer format, where the lower word is a fractional value.
121 */
122 inline Q16n16 Q16n16_mtof(Q16n16 midival_fractional)
123 {
124  Q16n16 diff_fraction;
125  uint8_t index = midival_fractional >> 16;
126  uint16_t fraction = (uint16_t) midival_fractional; // keeps low word
127  Q16n16 freq1 = (Q16n16) FLASH_OR_RAM_READ<const uint32_t>(MozziPrivate::MidiToFreq::midiToFreq + index);
128  Q16n16 freq2 = (Q16n16) FLASH_OR_RAM_READ<const uint32_t>((MozziPrivate::MidiToFreq::midiToFreq + 1) + index);
129  Q16n16 difference = freq2 - freq1;
130  if (difference>=65536)
131  {
132  diff_fraction = ((difference>>8) * fraction) >> 8;
133  }
134  else
135  {
136  diff_fraction = (difference * fraction) >> 16;
137  }
138  return (Q16n16) (freq1+ diff_fraction);
139 };
